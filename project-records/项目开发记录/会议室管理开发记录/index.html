时间轴组件封装 #

代码与展示效果 #

经过多个版本的修改，最终封装的组件如下：

vue
<template>
    <div class="timeLineSelectorWrapper">
        <div class="coordinateBox">
            <div
                v-if="selectorOptions.isShowDate"
                class="siteBox"
            ></div>
            <div class="coordinate">
                <div
                    v-for="index of 16"
                    :key="index"
                    class="itemBox"
                >
                    {{ index + 8 }}
                </div>
            </div>
        </div>
        <div
            v-for="appointTime in appointTimesData"
            :key="appointTime.date"
            class="timeLineSelectorRow"
        >
            <div
                v-if="selectorOptions.isShowDate"
                class="dateBox"
            >
                {{ dateShowFormatter(appointTime.date) }}
            </div>
            <div
                ref="$timeLineSelectorBoxRef"
                class="timeLineSelectorBox"
                @contextmenu.prevent="false"
            >
                <!-- 注意：此处的坐标是从1开始的 -->
                <div
                    v-for="index of 30"
                    :key="index"
                    :data-index="index"
                    :data-appointDate="appointTime.date"
                    :class="[
                        'itemBox',
                        appointTime.times.includes(index) ? 'closed' : '',
                        getOuterAppointTimes(appointTime.date).includes(index) ? 'selected' : '',
                    ]"
                    @mousedown="addEvent"
                    @mouseup="removeEvent"
                    @mouseout="mouseOutEvent"
                ></div>
                <slot
                    name="order"
                    :rowData="appointTime"
                ></slot>
            </div>
        </div>
        <div
            v-if="selectorOptions.requiredLenCheckTips"
            class="messageBox"
        >
            <p v-if="checkTips.maxLenCheck">提示：单次预约的最大可预约时长为5小时</p>
            <p v-if="checkTips.multipleMaxLenCheck">提示：同一会议室当日预约不可超过5小时，如有疑问请咨询阿霖</p>
        </div>
    </div>
</template>

<script lang="ts">
import dayjs from 'dayjs'
import _ from 'lodash'
import { Vue, Component, Prop, Emit, Ref } from 'vue-property-decorator'

interface IOptions {
    // 是否禁用所有选择功能
    disabled: boolean
    // 是否展示日期
    isShowDate: boolean
    // 是否支持选中closed块
    isSupportSelectClosed: boolean
    // 是否限制只能滑动连续的一段
    onlyOnce: boolean
    // 一次性最多选中块的长度
    maxLen: number
    // 多次最多选中块的个数
    multipleMaxLen: number
    // 是否需要长度校验提示
    requiredLenCheckTips: boolean
}
@Component({
    name: 'TimeLineSelector',
})
export default class extends Vue {
    // appointTimes是已经安排的时间，置灰
    @Prop({
        type: Array,
        required: true,
    }) appointTimes: Array<{ date: string, times: number[] }>
    @Prop({
        type: Array,
        required: false,
        default: () => [],
    }) selectedAppointTimes: Array<{ date: string, times: number[] }>
    @Prop({
        type: Object,
        required: false,
    }) options: Partial<IOptions>
    @Ref('$timeLineSelectorBoxRef') $timeLineSelectorBoxRef: Array<HTMLDivElement>
    @Emit('setAppointTime') setAppointTime() {
        return this.selectedData
    }
    mounted() {
        this.selectedData = this.selectedAppointTimes
    }
    mouseMinIndex = Number.MAX_SAFE_INTEGER
    mouseMaxIndex = Number.MIN_SAFE_INTEGER
    mouseCurIndex = -1
    mouseType: 'select'| 'unselect' = 'select'
    selectedData: Array<{ date: string, times: number[] }> = []
    checkTips = {
        maxLenCheck: false,
        multipleMaxLenCheck: false,
    }
    get selectorOptions():IOptions {
        return {
            disabled: false,
            isShowDate: false,
            isSupportSelectClosed: false,
            onlyOnce: false,
            maxLen: Infinity,
            multipleMaxLen: Infinity,
            requiredLenCheckTips: false,
            ...this.options,
        }
    }
    get appointTimesData() {
        if (_.isNil(this.appointTimes) || this.appointTimes.length === 0) {
            return [{ date: '', times: [] }]
        }
        return this.appointTimes
    }
    getOuterAppointTimes(date) {
        if (_.isNil(this.selectedAppointTimes) || this.selectedAppointTimes.length === 0) {
            return []
        }
        return this.selectedAppointTimes.find(e => e.date === date)?.times ?? []
    }
    // 中断鼠标移动操作
    fuseMouseOverEvents(node) {
        const parent = node.parentNode
        parent.childNodes.forEach(item => item.removeEventListener('mouseover', this.mouseOverCheck))
        this.mouseMaxIndex = Number.MIN_SAFE_INTEGER
        this.mouseMinIndex = Number.MAX_SAFE_INTEGER
    }
    // 鼠标移动操作中断条件判断
    judgeFuseEvent(e) {
        const judgeCrossCoordinate = e => {
            const curNode = e.target
            const outNode = e.relatedTarget
            const curNodeTop = curNode?.getBoundingClientRect()?.top
            const curNodeBottom = curNode?.getBoundingClientRect()?.bottom
            const outNodeTop = outNode?.getBoundingClientRect()?.top
            const outNodeBottom = outNode?.getBoundingClientRect()?.bottom
            return curNodeTop === outNodeTop && curNodeBottom === outNodeBottom
        }
        const judgeClosed = e => {
            if (this.selectorOptions.isSupportSelectClosed) { return true }
            const outNode = e.relatedTarget
            return !outNode.classList.contains('closed')
        }
        const judgeCrossMaxLen = () => {
            // 如果是取消选中，不使用该校验
            if (this.mouseType === 'unselect') {
                return true
            }
            const len = this.mouseMaxIndex - this.mouseMinIndex
            if (len < this.selectorOptions.maxLen - 1) {
                return true
            }
            this.checkTips.maxLenCheck = true
            return false
        }
        if (!(judgeCrossCoordinate(e) && judgeCrossMaxLen() && judgeClosed(e))) {
            // console.warn('触发熔断判断')
            this.fuseMouseOverEvents(e.target)
        }
    }
    addEvent(e) {
        if (this.selectorOptions.disabled) {
            return
        }
        const judgeOnlyOnce = () => {
            const isConsecutive = (arr: number[]) => {
                let maxNum = Number.MIN_SAFE_INTEGER
                let minNum = Number.MAX_SAFE_INTEGER
                arr.forEach(e => {
                    maxNum = Math.max(e, maxNum)
                    minNum = Math.min(e, minNum)
                })
                return maxNum - minNum === arr.length - 1
            }
            // 没有选择情况放行
            if (this.selectedData.length === 0) { return false }
            for (const item of this.selectedData) {
                if (!isConsecutive(item.times)) {
                    return false
                }
            }
            return true
        }
        if (this.selectorOptions.onlyOnce && judgeOnlyOnce()) {
            this.fuseMouseOverEvents(e.target)
            return
        }
        // 当前触发时间的节点
        const node = e.target
        const parent = node.parentNode
        /**
         * 选中态：变为取消选择模式
         * 未选中态：变为选择模式
         * 关闭态：禁止事件
         */
        if (!this.selectorOptions.isSupportSelectClosed && node.classList.contains('closed')) {
            return
        }
        if (node.classList.contains('selected')) {
            this.mouseType = 'unselect'
        } else {
            this.mouseType = 'select'
        }
        // 补充首元素
        this.mouseOverCheck(e)
        parent.childNodes.forEach(item => item.addEventListener('mouseover', this.mouseOverCheck))
    }
    removeEvent(e) {
        // 当前触发时间的节点
        const node = e.target
        this.fuseMouseOverEvents(node)
        this.setAppointTime()
    }
    mouseOutEvent(e) {
        this.judgeFuseEvent(e)
    }
    mouseOverCheck(e) {
        const judgeMultipleMaxLen = e => {
            const appointDate = e.target.dataset?.appointdate ?? ''
            const item = this.selectedData.find(e => e.date === appointDate)
            const times = item?.times?.map(e => Number(e)) ?? []
            return times.length < this.selectorOptions.multipleMaxLen
        }
        if (this.mouseType === 'select') {
            if (!judgeMultipleMaxLen(e)) {
                this.fuseMouseOverEvents(e.target)
                this.checkTips.multipleMaxLenCheck = true
                return
            } 
            this.checkTips.maxLenCheck = false
            this.checkTips.multipleMaxLenCheck = false
        }
        // 当前触发时间的节点
        const node = e.target
        const index = Number(node.dataset.index)
        // 触发日期
        const appointDate = node.dataset.appointdate ?? ''
        this.mouseCurIndex = index
        this.mouseMinIndex = Math.min(this.mouseMinIndex, this.mouseCurIndex)
        this.mouseMaxIndex = Math.max(this.mouseMaxIndex, this.mouseCurIndex)
        if (this.mouseType === 'select') {
            const item = this.selectedData.find(i => i.date === appointDate)
            if (_.isNil(item)) {
                this.selectedData.push({
                    date: appointDate,
                    times: [index],
                })
            } else {
                const set = new Set(item.times)
                set.add(index)
                this.selectedData = this.selectedData.map(i => {
                    if (i.date === appointDate) {
                        return {
                            date: i.date,
                            times: Array.from(set),
                        }
                    }
                    return i
                })
            }
            node.classList.add('selected')
        } else {
            const item = this.selectedData.find(i => i.date === appointDate)
            if (!_.isNil(item)) {
                const set = new Set(item.times)
                set.delete(index)
                this.selectedData = this.selectedData.map(i => {
                    if (i.date === appointDate) {
                        return {
                            date: i.date,
                            times: Array.from(set),
                        }
                    }
                    return i
                })
            }
            node.classList.remove('selected')
        }
    }
    dateShowFormatter(date: string): string {
        const weekMap = new Map([
            [0, '周日'],
            [1, '周一'],
            [2, '周二'],
            [3, '周三'],
            [4, '周四'],
            [5, '周五'],
            [6, '周六'],
        ])
        const weekDayStr = weekMap.get(dayjs(date).day())
        return `${dayjs(date)?.format('MM.DD')} (${weekDayStr})`
    }
    resetAllSelect() {
        this.selectedData = []
        this.$timeLineSelectorBoxRef.forEach(timeLineSelectorBox => {
            const children = timeLineSelectorBox.children
            for (const item of children) {
                item.classList.remove('selected')
            }
        })
    }
}
</script>

功能分析 #

组成元素：

  • 顶部坐标
  • 轴线主体（1~7行都有可能）
    • 左侧日期
    • 中间时间轴主体
    • 右侧预约按钮（只有一个页面才有）

数据源：

  • 之前已经选中过了
  • 滑动弹出的数据
  • 当前选中的数据

边界情况：

  • 滑动到上一行或下一行怎么办
  • 滑动遇到已经选中的块怎么办
  • 滑动超过10个怎么办

这个组件的设计是有败笔的，最大的问题的就是没有完全做到数据驱动视图，通过操作来改变数据，然后数据的变化触发视图的变化。该组件设计的问题在于在操作部分通过添加类名的方式修改视图，跳过了视图部分。但在某些情况下，其实视图是会限制操作的，因此想到另外一条视图限制操作。

归纳下来主要是理顺操作、数据、视图三种的关系，通过操作修改数据，数据的修改响应式的触发视图的更改，然后通过视图的更替限制用户操作行为。除了一些特殊情况，比如特别负责的交互如动画等等，都应该遵循这一原则。

确立了数据为核心，那么组件内部的一些数据一旦发生修改，那么就要响应式的更新组件外部的数据。

第二个遗憾的点就是该组件必须在某个限定宽度内视图才是正常的，最理想的方式应该是大于组件宽度时候让时间轴主体触发横向滚动效果，但由于组件设计原因，该效果并没有实现。
